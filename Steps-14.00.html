Create DAX measures 14
  1. Select the Walmart Table under the fields pane by clicking on it 
  2. From Power BI desktop ribbon, Home tab click on New measure 
  3. The formula bar appears along the top of the report canvas. 
  4.  Fill in all the DAX formulas provided below, one at a time, and when you finish Press Enter or select Commit (checkmark icon) in the formula bar.  

  5. After creating each measure choose the necessary format and decimal for each value (measure)
  6. Save

.Arrow YOY CPI = SWITCH(TRUE(), [.CPI YOY] <0, UNICHAR(9662),[.CPI YOY] =0, UNICHAR(9666), UNICHAR(9652))

.Arrow YOY% = SWITCH(TRUE(), [.Sales YOY] <0, UNICHAR(9662),[.Sales YOY] =0, UNICHAR(9666), UNICHAR(9652))

.AVG CPI = AVERAGE(WalmartSales[CPI_x])

.AVG Sales = AVERAGE(WalmartSales[Weekly_Sales])

.CPI YOY = 
VAR CurrentYearValue = [.AVG CPI]
VAR PreviousYearValue = CALCULATE([.AVG CPI], SAMEPERIODLASTYEAR('Calendar'[Date]))

RETURN
    IF(
        CurrentYearValue = 0,
        BLANK(),  -- or you can replace this with any other value or calculation
        DIVIDE(CurrentYearValue - PreviousYearValue, PreviousYearValue)
    )

.Sales Per Month = DIVIDE([.Total Sales],DISTINCTCOUNT('Calendar'[Month Name Short]),0)+0

.Sales YOY = 
VAR CurrentYearValue = [.AVG Sales]
VAR PreviousYearValue = CALCULATE([.AVG Sales], SAMEPERIODLASTYEAR('Calendar'[Date]))

RETURN
    IF(
        CurrentYearValue = 0,
        BLANK(),  -- or you can replace this with any other value or calculation
        DIVIDE(CurrentYearValue - PreviousYearValue, PreviousYearValue)
    )

.Total Sales = SUM(WalmartSales[Weekly_Sales])

AVG Holiday sales = AVERAGE(WalmartSales[holidaysales])

AVG non Holiday sales = AVERAGE(WalmartSales[Non holidaysales])

HolidayInc% = 
VAR HolidayPeriods = VALUES('WalmartSales'[Specific_US_Holiday])
VAR AverageNonHolidaySales = CALCULATE(
    AVERAGE('WalmartSales'[Non holidaysales]),
    ALL('WalmartSales'), // Removes any existing row context, but you might adjust this to respect certain filters.
    'WalmartSales'[Specific_US_Holiday] = "Non holiday" // Focuses on non-holiday sales.
)
RETURN
    AVERAGEX(
        HolidayPeriods,
        VAR HolidaySales = CALCULATE(
            AVERAGE('WalmartSales'[holidaysales]),
            'WalmartSales'[Specific_US_Holiday] = EARLIER('WalmartSales'[Specific_US_Holiday])
        )
        RETURN
            IF(
                HolidaySales <> 0,
                DIVIDE(HolidaySales - AverageNonHolidaySales, AverageNonHolidaySales),
                BLANK()
            )
    )

WeeklySales = CALCULATE(AVERAGE(WalmartSales[holidaysales]) + AVERAGE(WalmartSales[Non holidaysales])) / 2 

YoY Growth = 
VAR CurrentYearValue = [WeeklySales]
VAR PreviousYearValue = CALCULATE([WeeklySales], SAMEPERIODLASTYEAR('Calendar'[Date]))

RETURN
IF(
    ISBLANK(PreviousYearValue) || PreviousYearValue = 0,
    BLANK(),
    IF(
        (CurrentYearValue - PreviousYearValue) / PreviousYearValue = -1,
        BLANK(),
        (CurrentYearValue - PreviousYearValue) / PreviousYearValue
    )
)
